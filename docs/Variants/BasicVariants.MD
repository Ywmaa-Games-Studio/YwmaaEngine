# Here a quick list of the definitions used in this engine code for the basic C variable types

## Unsigned int types.

unsigned char: u8

unsigned short: u16

unsigned int: u32

unsigned long long: u64

## Signed int types.

signed char: i8

signed short: i16

signed int: i32

signed long long: i64

## Floating point types

float: f32

double: f64

## Boolean types

int: b32

char: b8
